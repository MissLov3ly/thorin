---
title: RadioButtonGroup
description: Allows for automatic state management of radio buttons.
---

```tsx
import { RadioButtonGroup } from '@ensdomains/thorin'
```

```tsx live=true
<Card shadow>
  <FieldSet
    legend={`Radio Buttons - Current Value: ${
      getState('radiobuttongroup') || '30'
    }`}
  >
    <RadioButtonGroup
      value={getState('radiobuttongroup') || '30'}
      onChange={(e) => setState('radiobuttongroup', e.target.value)}
    >
      <RadioButton label="Label 10" name="RadioButtonGroup" value="10" />
      <RadioButton label="Label 20" name="RadioButtonGroup" value="20" />
      <RadioButton label="Label 30" name="RadioButtonGroup" value="30" />
    </RadioButtonGroup>
  </FieldSet>
</Card>
```
