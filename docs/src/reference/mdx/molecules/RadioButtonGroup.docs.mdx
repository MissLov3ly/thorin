---
title: RadioButtonGroup
description: Allows for automatic state management of radio buttons.
---

```tsx
import { RadioButtonGroup } from '@ensdomains/thorin'
```

```tsx live=true
<Card padding="4" shadow>
  <FieldSet
    legend={`Radio Buttons - Current Value: ${
      getState('radiobuttongroup') || '30'
    }`}
  >
    <RadioButtonGroup onChange={(val) => setState('radiobuttongroup', val)}>
      <RadioButton label="Label" name="RadioButtonGroup" value="10" />
      <RadioButton label="Label" name="RadioButtonGroup" value="20" />
      <RadioButton checked label="Label" name="RadioButtonGroup" value="30" />
    </RadioButtonGroup>
  </FieldSet>
</Card>
```
