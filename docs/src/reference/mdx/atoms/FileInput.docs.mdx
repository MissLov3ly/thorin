---
title: FileInput
description: A HOC used to build components for accepting files.
---

```tsx
import { FileInput } from '@ensdomains/thorin'
```

```tsx live=true expand=true
<FileInput onChange={(file) => alert(file)}>
  {(context) =>
    context.name ? (
      <Stack alignItems="center" flexDirection="row">
        {context.name}
        <Button
          shape="circle"
          size="small"
          variant="secondary"
          onClick={context.reset}
        >
          <VisuallyHidden>Remove</VisuallyHidden>
          <CloseSVG />
        </Button>
      </Stack>
    ) : (
      <div>{context.droppable ? 'Drop file' : 'Attach file'}</div>
    )
  }
</FileInput>
```

## Max Size

```tsx live=true
<FileInput
  maxSize={0.01}
  onChange={(file) => alert(file)}
  onError={(error) => alert(error)}
>
  {(context) =>
    context.name ? (
      <div>
        {context.name}
        <Button
          shape="circle"
          size="small"
          variant="transparent"
          onClick={context.reset}
        >
          <VisuallyHidden>Remove</VisuallyHidden>
          <IconClose />
        </Button>
      </div>
    ) : (
      <div>{context.droppable ? 'Drop file' : 'Attach file (max 10kb)'}</div>
    )
  }
</FileInput>
```
