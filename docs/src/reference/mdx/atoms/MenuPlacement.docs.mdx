---
title: MenuPlacement
description: Provides the option of using a portal to append the menu to a different element. Useful when the menu is in a scrolling element whose overflow is hidden.
---

```tsx
import { MenuPlacement } from '@ensdomains/thorin'
```

```tsx
<MainElement ref={appendToRef}>
  <ScrollingElement ref={scrollingRef}>
    <ControlElement ref={controlRef}>
      <Button onClick={() => setMenuOpen(true)}>Open Menu</Button>
    </ControlElement>
    {menuOpen && (
      <MenuPlacement
        appendTo={appendToRef.current}
        control={controlRef.current}
        listenTo={scrollingRef.current}
      >
        Menu
      </MenuPlacement>
    )}
  </ScrollingElement>
</MainElement>
```
