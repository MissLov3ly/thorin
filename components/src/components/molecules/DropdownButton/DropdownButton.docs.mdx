---
title: DropdownButton
description: Creates a dropdown with a corresponding button
---

```tsx
import { DropdownButton } from '@ensdomains/thorin'
```

```tsx live=true expand=true minHeight=300px
<DropdownButton
  dropdownItems={[
    { label: 'Dashboard', onClick: () => null },
    { label: 'Disconnect', onClick: () => null, color: 'red' },
  ]}
>
  Open Menu
</DropdownButton>
```

<DropdownButton
